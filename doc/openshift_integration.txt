    - validate XML, iterate through accounts and spawn new thread for each account
    - invoke:  provisionOpenshiftAccount(String accountId, String password)
        - refreshDomain()

        - createPfpCoreApp()
            - include postgresql cartridge

refreshDomain()
    - NOTE:  only one domain allowed per user account
    - curl -k -X GET https://openshift.redhat.com/broker/rest/domains --user "rhn-gps-jbride:xxx"
        - parse for "id"
    - curl -k -X DELETE https://openshift.redhat.com/broker/rest/domains/jbride --user "rhn-gps-jbride:xxx" --data "force=true"
        - Force delete domain.  i.e. delete any applications under this domain
    - curl -k -X POST https://openshift.redhat.com/broker/rest/domains/ --user "rhn-gps-jbride:xxx" --data "id=pfpBRMS"
        - create a new domain


createBrmsWebsApp()
    - create medium size "brmsWebs" app
        curl -k -X POST https://openshift.redhat.com/broker/rest/domains/pfpBRMS/applications --user "rhn-gps-jbride:xxx" \
            --data "name=brmsWebs&cartridge=jbossas-7&scale=false&gear_profile=medium"
        - parse response for:
            git_url
        - write properties to a log file
    - add postgresql cartridge
        curl -k -X POST https://openshift.redhat.com/broker/rest/domains/pfpBRMS/applications/brmsWebs/cartridges --user "rhn-gps-jbride:xxx" \
            --data "cartridge=postgresql-8.4"
        - parse response for:  
            connection_url
            username
            password
        - write properties to a log file
    - execute $PFP_HOME/conf/postgresql/processFlowProvision.sql
